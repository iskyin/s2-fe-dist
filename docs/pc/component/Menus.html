# ··· Menus（菜单）

# 属性

参数 说明 类型 可选值 默认值
menu 菜单导航数组 Array<{ name: string; link: string; }> - []

# 基础用法


import Menus from "@/components/Menus";

<Menus menu={menuArr} />

# 实例


import Menus from "@/components/Menus";

  const menuArr = [
    {
      name: "银行账户统计分析",
      link: "/admin/account",
    },
    {
      name: "银行对账单管理",
      link: "/admin/statements",
    },
    {
      name: "余额调节表管理",
      link: "/admin/reconciliation",
    },
    {
      name: "借阅管理",
      link: "/admin/borrow",
    },
  ];

<Menus menu={menuArr} />
