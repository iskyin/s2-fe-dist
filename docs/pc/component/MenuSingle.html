# ··· MenuSingle（单级菜单）

说明： 单级菜单组件，是根据路由树的二级节点自动生成的

// 路由对象
const routes: Array<RouteConfig> = [
  {
    path: '/',
    redirect: '/home',
    name: '财经IT前端应用',
    component: Layout,
    children: [
      // ***** 菜单路由根据此级路由自动生成 *****
      {
        path: '/home',
        name: '主页',
        component: home,
      },
      {
        path: '/test',
        redirect: '/test/m1',
        name: '业务模块一',
        component: Noop,
        children: [
          {
            path: 'm1',
            name: '模块一子菜单一',
            component: m1,
          },
          {
            path: 'm2',
            name: '模块一子菜单二',
            component: m2,
          },
        ],
      }, // 业务模块1
      // ...  其他业务模块
    ],
  },
  {
    path: '*',
    name: '无效路由',
    redirect: '/home',
  },
  // ...  其他业务模块
];
