# ··· EchartLineBar（柱状折线图）

# 属性

参数 说明 类型 可选值 默认值 必传
eData 图表需要参数 Object - - Y
width 图表宽度 Number - 800 N
height 图表高度 Number - 500 N

# 参数详细说明

# ··· eData【图表需要参数】

参数 说明 类型 可选值 默认值 必传
title 标题 string - - Y
xName X 轴 名称 string - - Y
xData X 轴 数据 数组 number[] - - Y
barName 柱状图 标识 string - - Y
barUnit 柱状图 单位 string - - Y
barY 柱状图 数据 数组 number[] - - Y
lineName 折线图 标识 string - - Y
lineUnit 折线图 单位 string - - Y
lineY 折线图 单位 string - - Y

# 基础用法


<template>
    <EchartLineBar :eData="eData" :width="width" :height="height"></EchartLineBar>
</template>

import EchartLineBar from '@/components/EchartLineBar';
