# ··· Footer（底部）

# 属性

参数 说明 类型 可选值 默认值
l1 第一行 版权说明 String - ''
l2 第二行 版权说明 String - ''

# 基础用法


import Footer from "@/components/Footer";

<Footer l1={"版权第一行说明"} l2={"版权第二行说明"} />

# 实例


import Footer from "@/components/Footer";

<Footer
  l1={"Copyright © 1998 - 2021 Tencent. All Rights Reserved."}
  l2={"腾讯公司 版权所有"}
/>
