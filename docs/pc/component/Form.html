# ··· Form 表单查询

# 属性

参数 说明 类型 可选值 默认值
ArrForm Form的数据结构 String - -
column Form的列数 Number 3/2 -
onSuccess 表单验证成功后返回的函数 Function -
onError 表单验证失败后返回的函数 Function -

# 基础用法

    <FormSearch
        ArrForm={ArrForm}
        column={3}
        onSuccess={onSuccess}
        onError={onError}
    />

# 实例


    const ArrForm = reactive({
      width: 1800,
      FormColumn: [
        {name: 'v1', type: 'text' ,label: '账户编号', placeholder: '请输入XXX' ,rules: [{ required: false, message: 'Please input your username!' }]},
        {name: 'v2', type: 'text' ,label: '账户编号', placeholder: '请输入XXX' },
        {name: 'v3', type: 'select' ,label: '账户编号', placeholder: '请输入XXX', options: [{label: "label1", value: 'value1'},{label: "label2", value: 'value2'}] },
      ],
      button:[
        {text: "查询", color: 'red'}
      ],
    })


    <FormSearch
        ArrForm={ArrForm}
        column={3}
        onSuccess={onSuccess}
        onError={onError}
    />
