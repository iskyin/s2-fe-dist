# ··· PreviewPDFAndOFD（PDF、OFD 预览）

# 属性

参数 说明 类型 可选值 默认值 必传
fileData 文件数据 Object - - Y

# ··· fileData【文件数据】

参数 说明 类型 可选值 默认值 必传
fileType 文件类型 String ofd 或者 pdf - Y
fileContent 文件流数据 Base64 格式文件 - - Y
const fileData: Object = {
  fileType: 'ofd', // ofd或者pdf
  fileContent: 'UEsDBBQACAgIACKTDVEAAAAAAAAAAAAAAAASAAAARToMwFH3....', // Base64格式文件
};

# 基础用法

<template>
  <PreviewPDFAndOFD></PreviewPDFAndOFD>
</template>;

import PreviewPDFAndOFD from '@/components/Header';

  <PreviewPDFAndOFD :file-data="fileData" ></PreviewPDFAndOFD>
