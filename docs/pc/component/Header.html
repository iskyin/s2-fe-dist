# ··· Header（头部）

# 属性

参数 说明 类型 可选值 默认值 必传
logoSrc logo 地址 String - '' Y
companyName 平台名称 String - '' N
linkArr 头部导航栏数组 Array<{ name: string; link: string; }> - [] Y
menu 菜单区域插槽 slot - null Y

# 基础用法

<template>

    <Header :logoSrc='imgSrc' :companyName="companyName" :linkArr='headerArr'>
        <template v-slot:menu>
          // 菜单区域
        </template>
    </Header>

</template>

import Header from "@/components/Header";
