# ··· Header（头部）

# 属性

参数 说明 类型 可选值 默认值
logo logo地址 String - ''
link 头部导航栏数组 Array<{ name: string; link: string; }> - []
menu 菜单组件 Object - null

# 基础用法


import Header from "@/components/Header";

<Header logo={imgSrc} link={headerJson} menu={Menus} />
