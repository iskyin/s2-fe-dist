# 接口统一存放

规范要求：

  • src/api/url.ts文件中统一管理接口地址
  • 接口路径前缀可以添加/，因为接口处理时会过滤掉多个/情况
  • 接口中必须添加注释说明，方便其他合作人员调用

如下：

/**
 * 所有 url 都放在这里统一管理
 */
const ApiUrl = {
  //  登陆获取Tiket
  getLoginUrl: '/getLoginUrl',
  //  获取TOKEN
  getTokenByTicket: '/getTokenByTicket',

  // ... 其它接口
};

export default ApiUrl;

# 接口数据处理规范

规范要求：

  • 接口地址统一调用url.ts
  • AJAX 方法统一调用公共方法中封装的request
  • 接口中必须声明传入参数类型
  • 最后返回的数据是页面调用中可以直接使用的数据，在页面调用的地方不用再做任何处理
  • 在 api 函数里进行参数前置处理，直接返回接口需要的参数形式

如下：

import { post } from '@/common/request/index';
import ApiUrl from '@/api/url';

/**
 * 登陆获取Tiket
 * @param {
 *  sysId [1：登入，2登出]
 *  url [跳转的URL]
 * }
 * @return data [还有ticket的url]
 */
declare interface GetLoginUrlType<T = any> {
  sysId: number;
  url: string;
}
export async function getLoginUrl<T>(params: GetLoginUrlType) {
  return (await post(ApiUrl.getLoginUrl, params)) as any;
}

/**
 * 获取TOKEN
 * @param {
 *  ticket [TICKET]
 * }
 * @return data [组装的TOKEN对象]
 */
declare interface GetTokenByTicketType<T = any> {
  ticket: string;
}
export async function getTokenByTicket(params: GetTokenByTicketType) {
  return (await post(ApiUrl.getTokenByTicket, params)) as any;
}

# 视图层数据处理规范

规范要求：

  • 视图层主要负责交互逻辑实现，不建议对数据进行复杂二次处理
  • 视图层支持jsx 语言，但建议使用模板语法
  • 创建一个组件时共包含三个文件index.vue、index.less、index.ts
 
<template>
  <div class="body">
    <div class="body-box">
      模板语法
    </div>
  </div>
</template>

    'el-row': Row,
  },
})
export default class Child extends Vue {
  PageMenu = [] as any;
  // 业务逻辑
}
</script>
