# ··· Layout（布局）

网页整体框架布局

现有的布局分为上中下三个部分。

  • 设计稿内容宽度1200px；
  • 内容宽度1080px;

# 属性

参数 说明 类型 可选值 默认值
- - - - -

# 基础用法

import Layout from "@/components/Layout";

h(Layout)

# 实例


import layout from "@/components/Layout"; // 框架

const routes: Array<RouteRecordRaw> = [
  {
    path: "/",
    name: "调节表平台",
    component: layout,
    redirect: "/admin",
    children: [
      // ... other children router
    ],
  },
  // ... other router
];
